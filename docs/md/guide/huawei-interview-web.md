def get_ans(arr):
    n = len(arr)
    stack = []
    ans = [-1] * n
    for i in range(2 * n):
        j = i % n
        while len(stack) > 0 and arr[stack[-1]] < arr[j]:
            pos = stack.pop()
            ans[pos] = arr[j]
        stack.append(j)
    return ans
